HTML:

<ya-map ya-zoom="9" ya-center="[37.617761,55.755773]">
    <ya-geo-objects>
        <ya-geo-object ya-source="point"
                       ya-event-mouseenter="mouseenter($event)"
                       ya-event-mouseleave="mouseleave($event)"
                ></ya-geo-object>
    </ya-geo-objects>
</ya-map>

javascript:

$scope.point = {
    geometry:{
        type:'Point',
        coordinates:[37.617761,55.755773]
    }
};
$scope.mouseenter=function(e){
    e.get('target').options.set('preset', 'twirl#greenIcon');
};
$scope.mouseleave=function(e){
    e.get('target').options.unset('preset');
};
