HTML:

<ya-map zoom="9" center="[37.617761,55.755773]">
    <geo-objects>
        <geo-object source="point"
                    ya-mouseenter="mouseenter($event)"
                    ya-mouseleave="mouseleave($event)"
                ></geo-object>
    </geo-objects>
</ya-map>

javascript:

$scope.point = {
    geometry:{
        type:'Point',
        coordinates:[37.617761,55.755773]
    }
};
$scope.mouseenter=function(e){
    e.get('target').options.set('preset', 'twirl#greenIcon');
};
$scope.mouseleave=function(e){
    e.get('target').options.unset('preset');
};
