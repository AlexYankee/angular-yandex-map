HTML:

<ya-map id="map" zoom="8" center="[37.64,55.76]"></ya-map>

<label> Отслеживать размер
    <input ng-model="sizeChecked" type="checkbox">
</label>

javascript:

var timeout;
var resize = function(){
    if(timeout){
        $timeout.cancel(timeout);
    }
    timeout = $timeout(function(){
        var map = angular.element(document.getElementById('map'))
            .data('map');
        map.container.fitToViewport();
    },500);
};
$scope.$watch('sizeChecked',function(newValue,oldValue){
    if(newValue){
        angular.element($window).bind('resize', resize);
    }else if(oldValue){
        angular.element($window).unbind('resize',resize);
    }
});
